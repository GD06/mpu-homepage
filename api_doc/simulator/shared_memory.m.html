class SharedMemory:
    # (TODO)  Develop a more accurate timing model for multi-port and 
    # multi-bank shared memory 

    def __init__(self, env, log, clock_unit, capacity):
        self.env = env
        self.log = log 
        self.size = capacity 
        self.array = bytearray(capacity)
        self.clock_unit = clock_unit 

        self._read_req_queue = simpy.Store(env, capacity=1)
        self._read_resp_queue = simpy.FilterStore(env)
        self._write_req_queue = simpy.Store(env, capacity=1)

        # Spawn a process for read requests
        self.env.process(self._serve_read_request())

        # Spawn a process for write requests
        self.env.process(self._serve_write_request())

        return 

    def _serve_read_request(self):
        while True:
            req = yield self._read_req_queue.get() 
            assert isinstance(req, SMEMReadReq), "The incorrect type from " \
                "the request queue" 

            smem_data = self.array[
                req.smem_addr: req.smem_addr + req.data_size]
            resp = SMEMReadResp(
                smem_addr=req.smem_addr, 
                data_size=req.data_size, 
                data=smem_data,
            )

            yield self.env.timeout(1 * self.clock_unit) 
            yield self._read_resp_queue.put(resp) 
        return 

    def _serve_write_request(self):
        while True:
            req = yield self._write_req_queue.get()  
            assert isinstance(req, SMEMWriteReq), "The incorrect type from " \
                "the request queue"
            assert isinstance(req.data, bytearray), "The write transaction " \
                "to the shared memory should contain bytearray as the data"

            self.array[req.smem_addr: req.smem_addr + req.data_size] = req.data 
            yield self.env.timeout(1 * self.clock_unit) 
        return 

    def read_value(self, smem_addr_list, data_width):
        data_buffer = bytearray(0)
        for smem_addr in smem_addr_list:
            req = SMEMReadReq(smem_addr=smem_addr, data_size=data_width)
            yield self._read_req_queue.put(req)

            resp = yield self._read_resp_queue.get(
                lambda x: (x.smem_addr == smem_addr 
                           and x.data_size == data_width) 
            )
            data_buffer.extend(resp.data)

        return data_buffer

    def write_value(self, smem_addr_list, data_width, data_buffer, simt_mask):
        if simt_mask == 0:
            return 

        for i in range(len(smem_addr_list)):
            smem_addr = smem_addr_list[i]
            smem_data = data_buffer[i * data_width: (i + 1) * data_width]

            valid = (simt_mask >> i) & 1
            if valid:
                req = SMEMWriteReq(
                    smem_addr=smem_addr, 
                    data_size=data_width, 
                    data=smem_data
                )
                yield self._write_req_queue.put(req) 
        return
